## 二、迁移标准流程

### 2.1 新模块开发流程

```mermaid
graph LR
    A[分析Vue源文件] --> B[梳理API接口]
    B --> C[设计React组件]
    C --> D[实现列表页面]
    D --> E[实现表单功能]
    E --> F[实现详情页面]
    F --> G[功能测试]
    G --> H[代码优化]
```

### 2.2 详细步骤

#### 步骤1：分析Vue源文件
```bash
1. 定位Vue源文件
2. 分析页面结构
3. 理解业务逻辑
4. 识别组件依赖
5. 梳理数据流
```


### 3.2 带表单的列表页面

**适用场景**：标准CRUD功能

```jsx
import React, { useState } from 'react';
import { message, Form, Input } from 'antd';
import CmBasePage from '@components/CmBasePage';
import DataForm from '@components/DataForm';
import { getApiList, saveApi, updateApi, deleteApi } from '@api';

const CRUDPage = () => {
  const [formVisible, setFormVisible] = useState(false);
  const [editingRecord, setEditingRecord] = useState(null);
  const [loading, setLoading] = useState(false);

  const columns = [
    { title: '名称', dataIndex: 'name', key: 'name', width: 200 },
  ];

  const searchFieldList = [
    {
      title: '名称',
      dataIndex: 'name',
      key: 'SEARCH_LIKE_name',
      type: 'text',
    },
  ];

  const handleAdd = () => {
    setEditingRecord(null);
    setFormVisible(true);
  };

  const handleEdit = (record) => {
    setEditingRecord(record);
    setFormVisible(true);
  };

  const handleDelete = async (record) => {
    try {
      const response = await deleteApi({ id: record.id });
      if (response.code === 0 || response.code === 200) {
        message.success('删除成功');
        refreshTable();
      }
    } catch (error) {
      message.error('删除失败');
    }
  };

  const handleSubmit = async (values) => {
    setLoading(true);
    try {
      const response = editingRecord
        ? await updateApi(values)
        : await saveApi(values);

      if (response.code === 0 || response.code === 200) {
        message.success('操作成功');
        setFormVisible(false);
        refreshTable();
      }
    } finally {
      setLoading(false);
    }
  };

  const refreshTable = () => {
    document.querySelector('.ant-pro-table')?.parentNode?.dispatchEvent(
      new CustomEvent('reload')
    );
  };

  return (
    <div>
      <CmBasePage
        columns={columns}
        apiEndpoint="/api/endpoint"
        searchFieldList={searchFieldList}
        onAdd={handleAdd}
        onEdit={handleEdit}
        onDelete={handleDelete}
        rowKey="id"
      />

      <DataForm
        visible={formVisible}
        title={editingRecord ? '编辑' : '新增'}
        initialValues={editingRecord || {}}
        extraValues={{ id: editingRecord?.id }}
        onCancel={() => setFormVisible(false)}
        onSubmit={handleSubmit}
        onClosed={() => setEditingRecord(null)}
        loading={loading}
      >
        <Form.Item name="name" label="名称" rules={[{ required: true }]}>
          <Input placeholder="请输入名称" />
        </Form.Item>
      </DataForm>
    </div>
  );
};

export default CRUDPage;
```

### 3.3 带分类树的列表页面

**适用场景**：左侧分类树 + 右侧列表

```jsx
import React, { useState, useRef, useEffect } from 'react';
import CmBasePage from '@components/CmBasePage';
import CategoryTree from './components/CategoryTree';

const CategoryPage = () => {
  const [sidebarCollapsed, setSidebarCollapsed] = useState(false);
  const [selectedCategoryId, setSelectedCategoryId] = useState(null);
  const categoryIdRef = useRef(null);

  useEffect(() => {
    categoryIdRef.current = selectedCategoryId;
  }, [selectedCategoryId]);

  const loadData = async (params) => {
    if (categoryIdRef.current) {
      const searchParam = params.searchParam
        ? JSON.parse(params.searchParam)
        : {};
      searchParam.SEARCH_EQ_categoryId = categoryIdRef.current;
      params.searchParam = JSON.stringify(searchParam);
    }

    const response = await getDataList(params);
    return {
      data: response.data?.records || [],
      success: true,
      total: response.data?.total || 0
    };
  };

  const handleCategorySelect = (categoryId) => {
    setSelectedCategoryId(categoryId);
    setTimeout(() => {
      document.querySelector('.ant-pro-table')?.parentNode?.dispatchEvent(
        new CustomEvent('reload')
      );
    }, 0);
  };

  return (
    <div className="page-container">
      {!sidebarCollapsed && (
        <div className="page-sidebar">
          <CategoryTree onSelect={handleCategorySelect} />
        </div>
      )}

      <div
        className="sidebar-toggle-btn"
        onClick={() => setSidebarCollapsed(!sidebarCollapsed)}
        style={{ left: sidebarCollapsed ? 0 : 280 }}
      >
        {sidebarCollapsed ? '»' : '«'}
      </div>

      <div className="page-content">
        <CmBasePage columns={columns} onLoadData={loadData} rowKey="id" />
      </div>
    </div>
  );
};

export default CategoryPage;
```

## 四、常见问题处理

### 4.1 数据不回显问题

**问题**：编辑时表单数据不显示

**原因**：initialValues只在组件初始化时生效

**解决方案**：
```jsx
// ❌ 错误写法
<DataForm
  visible={visible}
  initialValues={editingRecord}  // 不会更新
/>

// ✅ 正确写法
<DataForm
  visible={visible}
  initialValues={editingRecord || {}}  // 确保有默认值
  key={editingRecord?.id}  // 添加key强制重新渲染
/>
```

### 4.2 表格不刷新问题

**问题**：操作后表格数据不更新

**解决方案**：
```jsx
// 使用自定义事件刷新
const refreshTable = () => {
  document.querySelector('.ant-pro-table')?.parentNode?.dispatchEvent(
    new CustomEvent('reload')
  );
};

// 在需要刷新的地方调用
refreshTable();
```

### 4.3 搜索参数格式问题

**问题**：搜索参数格式不正确

**解决方案**：
```jsx
// 确保使用正确的搜索参数格式
const searchFieldList = [
  {
    title: '名称',
    dataIndex: 'name',
    key: 'SEARCH_LIKE_name',  // 必须以 SEARCH_ 开头
    type: 'text',

];

// 后端会收到
{
  searchParam: '{"SEARCH_LIKE_name":"搜索值"}'
}
```

### 4.4 日期格式问题

## 五、代码优化建议

### 5.1 使用useMemo优化性能

```jsx
// 缓存计算结果
const options = useMemo(() => {
  return list.map(item => ({
    label: item.name,
    value: item.id,
  }));
}, [list]);

// 缓存树形数据
const treeData = useMemo(() => {
  return normalizeTreeData(rawData);
}, [rawData]);
```

### 5.2 使用useCallback缓存函数

```jsx
const handleSelect = useCallback((id) => {
  setSelectedId(id);
}, []);

const handleSubmit = useCallback(async (values) => {
  // 提交逻辑
}, [依赖项]);
```

### 5.3 使用ref避免闭包陷阱

```jsx
// ❌ 错误：闭包问题
useEffect(() => {
  const handler = () => {
    console.log(count);  // 始终是初始值
  };
  document.addEventListener('click', handler);
  return () => document.removeEventListener('click', handler);
}, []);

// ✅ 正确：使用ref
const countRef = useRef(count);
useEffect(() => {
  countRef.current = count;
}, [count]);

useEffect(() => {
  const handler = () => {
    console.log(countRef.current);  // 最新值
  };
  document.addEventListener('click', handler);
  return () => document.removeEventListener('click', handler);
}, []);
```

### 5.4 提取重复逻辑

```jsx
// ❌ 错误：重复代码
const handleSubmit1 = async (values) => {
  setLoading(true);
  try {
    const response = await api1(values);
    if (response.code === 0) {
      message.success('成功');
      setFormVisible(false);
      refreshTable();
    }
  } finally {
    setLoading(false);
  }
};

const handleSubmit2 = async (values) => {
  setLoading(true);
  try {
    const response = await api2(values);
    if (response.code === 0) {
      message.success('成功');
      setFormVisible(false);
      refreshTable();
    }
  } finally {
    setLoading(false);
  }
};

// ✅ 正确：提取公共逻辑
const useSubmitHandler = (apiFn) => {
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (values) => {
    setLoading(true);
    try {
      const response = await apiFn(values);
      if (response.code === 0) {
        message.success('成功');
        setFormVisible(false);
        refreshTable();
      }
    } finally {
      setLoading(false);
    }
  };

  return [handleSubmit, loading];
};
```
