# Vue到React快速参考手册

## 一、模板速查

### 1.1 完整页面模板

```jsx
import React, { useState, useRef, useEffect } from 'react';
import { message, Form, Input, Select, Tag } from 'antd';
import CmBasePage from '@components/CmBasePage';
import DataForm from '@components/DataForm';
import { getApiList, saveApi, updateApi, deleteApi } from '@api';

/**
 * 模块名称
 */
const ModulePage = () => {
  // === 状态声明 ===
  const [formVisible, setFormVisible] = useState(false);
  const [editingRecord, setEditingRecord] = useState(null);
  const [loading, setLoading] = useState(false);

  // === 列配置 ===
  const columns = [
    {
      title: '字段名称',
      dataIndex: 'fieldName',
      key: 'fieldName',
      width: 150,
    },
    {
      title: '状态',
      dataIndex: 'status',
      key: 'status',
      width: 100,
      render: (status) => {
        const map = {
          '0': { color: 'green', text: '正常' },
          '1': { color: 'orange', text: '停用' },
        };
        const info = map[status] || { color: 'default', text: '未知' };
        return <Tag color={info.color}>{info.text}</Tag>;
      },
    },
  ];

  // === 搜索配置 ===
  const searchFieldList = [
    {
      title: '字段名称',
      dataIndex: 'fieldName',
      key: 'SEARCH_LIKE_fieldName',
      type: 'text',
    },
    {
      title: '状态',
      dataIndex: 'status',
      key: 'SEARCH_EQ_status',
      type: 'select',
      options: [
        { label: '正常', value: '0' },
        { label: '停用', value: '1' },
      ],
    },
  ];

  // === 事件处理 ===
  const handleAdd = () => {
    setEditingRecord(null);
    setFormVisible(true);
  };

  const handleEdit = (record) => {
    setEditingRecord(record);
    setFormVisible(true);
  };

  const handleDelete = async (record) => {
    try {
      const response = await deleteApi({ id: record.id });
      if (response.code === 0 || response.code === 200) {
        message.success('删除成功');
        refreshTable();
      } else {
        message.error(response.msg || '删除失败');
      }
    } catch (error) {
      message.error('删除失败: ' + error.message);
    }
  };

  const handleSubmit = async (values) => {
    setLoading(true);
    try {
      const response = editingRecord
        ? await updateApi(values)
        : await saveApi(values);

      if (response.code === 0 || response.code === 200) {
        message.success(editingRecord ? '更新成功' : '创建成功');
        setFormVisible(false);
        refreshTable();
      } else {
        message.error(response.msg || '操作失败');
      }
    } catch (error) {
      message.error('操作失败: ' + error.message);
    } finally {
      setLoading(false);
    }
  };

  // === 刷新表格 ===
  const refreshTable = () => {
    document.querySelector('.ant-pro-table')?.parentNode?.dispatchEvent(
      new CustomEvent('reload')
    );
  };

  // === 渲染 ===
  return (
    <div>
      <CmBasePage
        columns={columns}
        apiEndpoint="/api/endpoint"
        apiMethod="post"
        searchFieldList={searchFieldList}
        onAdd={handleAdd}
        onEdit={handleEdit}
        onDelete={handleDelete}
        rowKey="id"
      />

      <DataForm
        visible={formVisible}
        title={editingRecord ? '编辑' : '新增'}
        initialValues={editingRecord || {}}
        extraValues={{ id: editingRecord?.id }}
        onCancel={() => setFormVisible(false)}
        onSubmit={handleSubmit}
        onClosed={() => setEditingRecord(null)}
        loading={loading}
        formType="drawer"
      >
        <Form.Item
          name="fieldName"
          label="字段名称"
          rules={[{ required: true, message: '请输入字段名称' }]}
        >
          <Input placeholder="请输入字段名称" />
        </Form.Item>

        <Form.Item
          name="status"
          label="状态"
          rules={[{ required: true, message: '请选择状态' }]}
        >
          <Select placeholder="请选择状态">
            <Select.Option value="0">正常</Select.Option>
            <Select.Option value="1">停用</Select.Option>
          </Select>
        </Form.Item>
      </DataForm>
    </div>
  );
};

export default ModulePage;
```

## 二、Vue组件速查

### 2.1 条件渲染

| Vue | React |
|-----|-------|
| `v-if="condition"` | `{condition && <Component />}` |
| `v-else` | `{!condition && <Component />}` |
| `v-show="condition"` | `style={{display: condition ? 'block' : 'none'}}` |

### 2.2 列表渲染

| Vue | React |
|-----|-------|
| `v-for="item in list"` | `{list.map(item => <Item key={item.id} />)}` |
| `:key="item.id"` | `key={item.id}` |

### 2.3 事件绑定

| Vue | React |
|-----|-------|
| `@click="handler"` | `onClick={handler}` |
| `@change="handler"` | `onChange={handler}` |
| `@submit.prevent="handler"` | `onSubmit={(e) => { e.preventDefault(); handler(); }}` |

### 2.4 表单绑定

| Vue | React |
|-----|-------|
| `v-model="value"` | `value={value} onChange={e => setValue(e.target.value)}` |
| `v-model.number="value"` | 使用Form.Item的name属性 |

### 2.5 样式绑定

| Vue | React |
|-----|-------|
| `:class="{active: isActive}"` | `className={isActive ? 'active' : ''}` |
| `:style="{color: color}"` | `style={{color: color}}` |
| `:class="classVar"` | `className={classVar}` |

### 2.6 属性绑定

| Vue | React |
|-----|-------|
| `:attr="value"` | `attr={value}` |
| `src="url"` | `src={url}` |
| `href="url"` | `href={url}` |

// 获取文件扩展名
export const getFileExt = (filename) => {
  return filename.slice(((filename.lastIndexOf('.') - 1) >>> 0) + 2);
};

// 文件大小格式化
export const formatFileSize = (bytes) => {
  if (bytes === 0) return '0 B';
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return (bytes / Math.pow(k, i)).toFixed(2) + ' ' + sizes[i];
};
